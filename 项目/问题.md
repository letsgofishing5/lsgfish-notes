    /**
 * Map转换为Object
 * @param {Map} map 输入map
 * @returns {Object}
 */
export function mapToObject(map) {
  return [...map.entries()].reduce((obj, [key, value]) => (obj[key] = value, obj), {})
}

/**
 * Map转换Array
 * @param {Map} map 输入map
 * @returns {[Object]}
 */
export function mapToObjectArray(map) {
  return [...map.entries()].map(item => ({ [item[0]]: item[1] }))
}

/**
 * Map转换Array {k: key, v: value}
 * @param {Map} map 输入map
 * @returns {[Object]}
 */
export function mapToKVArray(map) {
  return [...map.entries()].map(item => ({ k: item[0], v: item[1] }))
}
